func add(a: int, b: int): int{
	var c: int = a + b
	return c
}

func sqrDist(x: int, y: int): int{
	var dist: int = x * x + y * y
	return dist
}

	var a: int = (4 - 3) * (10 / 3)
	a = a + 1

	printInt(sum)

	printInt(sqrDist(3, 3))

}
